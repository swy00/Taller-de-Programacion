          dato: integer;

          HD: arbol;
         end;

Procedure preOrden (a:arbol);
begin
   if ( a <> nil ) then 
     begin
      write (a^.dato);
      preOrden (a^.HI);
      preOrden (a^.HD);
   end;
end;


Procedure postOrden (a:arbol);
begin
   if ( a <> nil ) then 
     begin
      postOrden (a^.HI);
      postOrden (a^.HD);
      write (a^.dato);
   end;
end;
